void find_prime(ll n){
    prime.clear();
    bool Prime[n];
    for(int i=0;i<=n;i++)Prime[i]=true;
    for(int i=2;i<=sqrt(n);i++){
        for(int j=i;i*j<=n;j++){
            Prime[i*j]=false;;
        }
    }
    for(int i=2;i<=n;i++){
        if(Prime[i])	prime.pb(i);
    }
    return;
}
vector<ll> base(ll a, ll b){
    vector<ll> c;
    while(a){
        c.pb(a%b);
        a/=b;
    }
    return c;
}

vector<pair<ll,ll> > decompose_prime(ll a){
    vector<pair<ll,ll> > c;
    ll i=0;
    while(a>1){
        
        if(i<prime.size()){
            if(a%prime[i])  {
            i++;
            continue;
        }
        c.pb(mk(prime[i],0));
        while(a%prime[i]==0){
            c[c.size()-1].s++;
            a/=prime[i];
        }
        }else{
            c.pb(mk(a,1));
            break;
        }
    }
    return c;
}
